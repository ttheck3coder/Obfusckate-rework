
start_spy_network = {

    is_shown = {
        OR = {NOT = {has_variable = spy_network}
        has_variable = pause_spy_network
        spy_network_value < 0}

        is_ai = no
        is_alive = yes
        is_adult = yes
        is_incapable = no
        is_landless_adventurer = no
    }
    is_valid = {
        spy_network_value > 0
    }
    effect = {
        remove_variable = pause_spy_network
        if = {
            limit = {NOT = {has_variable = spy_network}}
            set_variable = {
                name = spy_network
                value = 1.1 #0 game_start 没有spy_network
            }
        }
    }
}

stop_spy_network = {

    is_shown = {
        NOT = { has_variable = pause_spy_network }
        is_ai = no
        is_alive = yes
        is_adult = yes
        is_incapable = no
        is_landless_adventurer = no
        has_variable = spy_network
    }
    is_valid = {
        spy_network_value > 0
    }
    effect = {
        set_variable = {
            name = pause_spy_network
        }
    }
}
